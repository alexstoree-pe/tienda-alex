function calcularDescuento() { if (carrito.length < 2) return 0; let descuentoTotal = 0; // Ordenamos para que la plataforma más cara sea la "base" y no se le aplique descuento a ella // Esto asegura que el cliente reciba el mejor beneficio en combos mixtos const carritoOrdenado = [...carrito].sort((a, b) => b.precio - a.precio); for (let i = 1; i < carritoOrdenado.length; i++) { const itemActual = carritoOrdenado[i]; const itemAnterior = carritoOrdenado[i-1]; // Lógica para la SEGUNDA plataforma (Índice 1) if (i === 1) { if (itemActual.precio >= 10 && itemAnterior.precio >= 10) { descuentoTotal += 2; } else { // Si una es mayor a 10 y la otra menor, o ambas menores descuentoTotal += 1; } } // Lógica para la TERCERA plataforma en adelante (Índice 2, 3...) else { if (itemActual.precio >= 10) { descuentoTotal += 2; } else { descuentoTotal += 1; } } } return descuentoTotal; }
